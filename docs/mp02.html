load_atus_data <- function(file_base = c("resp", "rost", "act")){
    if(!dir.exists(file.path("data", "mp02"))){
        dir.create(file.path("data", "mp02"), showWarnings=FALSE, recursive=TRUE)
    }
    
    file_base <- match.arg(file_base)
    
    file_name_out <- file.path("data", 
                               "mp02", 
                               glue("atus{file_base}_0324.dat"))
        
    if(!file.exists(file_name_out)){
        
        url_end <- glue("atus{file_base}-0324.zip")
        
        temp_zip <- tempfile(fileext=".zip")
        temp_dir <- tools::file_path_sans_ext(temp_zip)
        
        request("https://www.bls.gov") |>
            req_url_path("tus", "datafiles", url_end) |>
            req_headers(`User-Agent` = "Mozilla/5.0 (Macintosh; Intel Mac OS X 10.15; rv:149.0) Gecko/20100101 Firefox/149.0") |>
            req_perform(temp_zip)
    
        unzip(temp_zip, exdir = temp_dir)
        
        temp_file <- file.path(temp_dir, basename(file_name_out))
        
        file.copy(temp_file, file_name_out)
    }
    
    file <- read_csv(file_name_out, 
                     show_col_types = FALSE)
    
    switch(file_base, 
           resp=file |> 
                  rename(survey_weight = TUFNWGTP, 
                         time_alone = TRTALONE, 
                         survey_year = TUYEAR),
           rost=file |>
                  mutate(sex = case_match(TESEX, 1 ~ "M", 2 ~ "F"),
                         age = TEAGE), 
           act =file |>
             rename(activity_location = TEWHERE) |>   
             mutate(activity_n_of_day = TUACTIVITY_N, 
                       time_spent_min = TUACTDUR24, 
                       start_time = TUSTARTTIM,
                       stop_time = TUSTOPTIME, 
                       level1_code = paste0(TRTIER1P, "0000"), 
                       level2_code = paste0(TRTIER2P, "00"),
                       level3_code = TRCODEP,
                       )
                      
           ) |> 
                rename(participant_id = TUCASEID) |>
                select(matches("[:lower:]", 
                               ignore.case=FALSE))
}

load_atus_activities <- function(){
    if(!dir.exists(file.path("data", "mp02"))){
        dir.create(file.path("data", "mp02"), showWarnings=FALSE, recursive=TRUE)
    }
    
    dest_file <- file.path("data",
                           "mp02",
                           "atus_activity_codes.csv")
    
    if(!file.exists(dest_file)){
        download.file("https://michael-weylandt.com/STA9750/mini/atus_activity_codes.csv", 
                      quiet=TRUE, 
                      destfile=dest_file, 
                      mode="wb")
    }
    
    read_csv(dest_file, show_col_types = FALSE) 
}

1st target demographic: Stay-at-home dads

Stay-at-home dads are defined as men with at least one child under the age of 18 in the household who are not employed full-time and are not enrolled in school. This group represents fathers whose primary responsibilities are centered around childcare and household management.

Based on data from the American Time Use Survey (ATUS) from 2020 through 2024, stay-at-home dads devote a considerable portion of their daily time to socializing, relaxing, and leisure activities. The median leisure time is approximately 3.5 hours per day, with more than half (53.5%) exceeding the 3-hour target, indicating that leisure time is relatively abundant within this group. While leisure supports overall well-being, this pattern suggests opportunities to better optimize time toward more health-oriented or productive activities.

A time-tracking application could help stay-at-home dads monitor their routines, set goals, and maintain a more balanced and intentional use of time.

  filter(
