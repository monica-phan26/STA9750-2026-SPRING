state_to_state_migration_2024 <- function() {
  
#Create the data folder
dir.create("data/mp03", recursive = TRUE, showWarnings = FALSE)

#Download the file 2024 if only it does not already exist

file_url <- "https://www2.census.gov/programs-surveys/demo/tables/geographic-mobility/2024/state-to-state-migration/State_to_State_Migration_Table_2024_T13.xlsx"

file_path <- "data/mp03/migration_2024.xlsx"

if(!file.exists(file_path)) {
  download.file(file_url, file_path, mode = "wb")
}

#Read the "table" sheet and rename columns, remove the last column.

moving_flow_raw <- read_excel(
  file_path,
  sheet = "Table",
  skip = 8,
  col_names = c("state_current", "state_1y", "population", "moe")
  )|>
  select(state_current, state_1y, population)

#Clean the "table" sheet

moving_flows <- moving_flow_raw |>
  filter(
    state_current %in% state.name,
    state_1y %in% state.name,
    state_current != state_1y
  ) |>
  mutate(
    population = recode(population, "N" = "0"),
    population = as.integer(gsub(",", "", population))
  ) |>
  transmute(
    state_current,
    state_1y,
    population,
    year_current = 2024,
    year_1y = 2023
  )

# Read "Supplemental - Current Res" sheet
stationary_flows <- read_excel(
  file_path,
  sheet = "Supplemental - Current Res",
  range = "A11:F100",
  col_names = FALSE,
  col_types = "text"
) |>
  select(
    state_current = 1,
    same_house = 4,
    same_state = 6
  ) |>
  mutate(
    state_current = str_squish(state_current),
    same_house = as.integer(gsub(",", "", same_house)),
    same_state = as.integer(gsub(",", "", same_state))
  ) |>
  filter(state_current %in% state.name) |>
  mutate(
    population = same_house + same_state
  ) |>
  transmute(
    state_current,
    state_1y = state_current,
    population,
    year_current = 2024,
    year_1y = 2023
  )

# Combine population from the moving flow and stationary flow

migration_2024 <- bind_rows(moving_flows, stationary_flows)

return(migration_2024)
}

state_to_state_migration_2023 <- function() {
  
#Create the data folder
dir.create("data/mp03", recursive = TRUE, showWarnings = FALSE)

#Download the file 2023 if only it does not already exist

file_url <- "https://www2.census.gov/programs-surveys/demo/tables/geographic-mobility/2023/state-to-state-migration/State_to_State_Migration_Table_2023_T13.xlsx"

file_path <- "data/mp03/migration_2023.xlsx"

if(!file.exists(file_path)) {
  download.file(file_url, file_path, mode = "wb")
}

# State name with District of Columbia:

states_with_dc <- c(state.name[1:8],
                    "District of Columbia",
                    state.name[9:50])

# Read the first 6 columns for stationary flows

stationary_flows <- read_excel(
  file_path,
  range = "A9:F100",
  col_names = FALSE,
  col_types = "text"
) |>
  select(
    state_current = 1,
    same_house = 4,
    same_state = 6) |>
  mutate(
    state_current = str_squish(state_current),
    same_house = as.integer(gsub(",", "", same_house)),
    same_state = as.integer(gsub(",", "", same_state))
  ) |>
  filter(state_current %in% state.name) |>
  mutate(
    population = same_house + same_state) |>
  transmute(
    state_current,
    state_1y = state_current,
    population = as.integer(population),
    year_current = 2023,
    year_1y = 2022
  )
  
# Read column J to DG for state_to_state moving flows
moving_raw <- read_excel(
  file_path,
  range = "A9:DG100",
  col_names = c(
    "state_current",
    rep(NA,8),
    vctrs::vec_interleave(states_with_dc, NA)
  ),
  col_types = "text"
  )|>
  select(
    state_current,
         any_of(states_with_dc)
    ) |>
  mutate(
    state_current = str_squish(state_current)
  )

moving_flows <- moving_raw |>
  pivot_longer(
    cols = all_of(states_with_dc),
    names_to = "state_1y",
    values_to = "population"
  ) |>
  filter(
    state_current %in% state.name,
    state_1y %in% state.name,
    state_current != state_1y
  ) |>
  mutate(
    population = as.integer(gsub(",", "", population))
  ) |>
  transmute(
    state_current,
    state_1y,
    population,
    year_current = 2023,
    year_1y = 2022
  )

# Combine stationary flows and moving flows:
migration_2023 <- bind_rows(
  moving_flows,
  stationary_flows
)

return(migration_2023)
}

# Function to move the 1st row to the column name:
move_row_to_colnames <- function(X, row = 1) {
  X <- as.data.frame(X)
  
  Xrow <- X[row, ]
  X <- X[-row, ]
  
  colnames(X) <- Xrow
  
  X |>
    type_convert() |>
    as_tibble()
}

# Function to pull the state name from the metro name:

pull_state_from_metro <- function(metro_name) {
  str_extract(
    metro_name,
    ".*, (\\S{2})[-[:alpha:]]* Metro Area",
    group = 1
  )
}

# Function
metro_to_metro_migration <- function() {
  #Create folder:
  dir.create("data/mp03", recursive = TRUE, showWarnings = FALSE)
  
  #Load the file using API URL
  api_url  <- paste0(
    "https://api.census.gov/data/2020/acs/flows?",
    "get=MOVEDIN,MOVEDOUT,FULL1_NAME,FULL2_NAME",
    "&for=metropolitan%20statistical%20area/micropolitan%20statistical%20area:*"
  )
  
# Local file path
file_path <- "data/mp03/metro_to_metro_migration_2020.json"

# Download JSON if needed
  if (!file.exists(file_path)) {
    download.file(api_url, file_path, mode = "wb")
  }
  
  # Read JSON
  metro_raw <- jsonlite::fromJSON(file_path)
  
  # Move first row to column names
  metro_clean <- move_row_to_colnames(metro_raw)
  
  # Clean final table
  metro_flows <- metro_clean |>
    select(
      moved_in = MOVEDIN,
      moved_out = MOVEDOUT,
      metro_current = FULL1_NAME,
      metro_1y = FULL2_NAME
    ) |>
    mutate(
      moved_in = as.integer(moved_in),
      moved_out = as.integer(moved_out),
      metro_current_state = pull_state_from_metro(metro_current),
      metro_1y_state = pull_state_from_metro(metro_1y)
    )
  
  return(metro_flows)
}

Proposal

The main goal of this strategy is to help New Jersey gain one more congressional seat after the 2030 Census. Based on the analysis above, the state is expected to grow by about 1.57 million people from 2020 to 2030. But just growing is not enough, what really matters is how New Jersey grows compared to other states. To make this work, we need to focus on bringing more people in while also reducing the number of people leaving.

A big opportunity comes from New York, which is by far the largest source of people moving into New Jersey. In 2024 alone, New Jersey gained about 20,000 more people from New York than it lost, which is much higher than any other state. This shows there is already a strong connection, especially with people living in NYC, and we should lean into that. The idea is to show New Yorkers that they can have a better lifestyle in New Jersey—more space, less crowding, and easier driving while still keeping access to NYC jobs, healthcare, and everything the metro area offers. Since many of these residents use public transportation every day, placing ads in subways and transit hubs would be very effective. Slogan: “New York on the West Side.”

At the same time, New Jersey is losing a lot of people to Florida, with over 10,000 net residents leaving in 2024. It’s difficult to compete with Florida on taxes, but New Jersey has an advantage when it comes to job opportunities, especially outside Florida’s major cities like Miami and Tampa. Many smaller cities in Florida offer fewer career options, which gives us a chance to bring people back. We can target these areas with ads in gas stations and highway stops, where people are more likely to notice them. The message should highlight that New Jersey offers a similar spacious, driving-friendly lifestyle, but with stronger job prospects and more variety in seasons and activities. Slogan: “Same Space. More Opportunity. Four Seasons to Enjoy.”

We should also look at states like Michigan, which are expected to keep the same number of seats. This means even small population changes could make a difference. New Jersey can attract people from Michigan by highlighting its strong job market, higher income potential, and proximity to major cities like NYC and Philadelphia. It also offers lifestyle benefits such as coastal access and more diverse communities. Slogan: “Closer to Opportunity. Connected to Everything.”

In summary, even though New Jersey is already growing, gaining another congressional seat will depend on making smart, targeted migration decisions. By focusing on strong inflow areas like New York, reducing losses to Florida, and competing with states like Michigan, New Jersey can realistically achieve the additional population needed by 2030 to secure one more congressional seat.
